 19 /**
 20  * Compute the Levenshtein distance between two strings.
 21  *
 22  * @see Source: <a href="http://en.wikibooks.org/wiki/Algorithm_Implementation/Strings/Levenshtein_distance#JavaScript">Wikibooks:Algorithm Implementation/Strings/Levenshtein distance</a>
 23  * @see Wikipedia: <a href="http://en.wikipedia.org/wiki/Levenshtein_distance">Levenshtein distance</a>
 24   * @public
 25  * @param {String} str1 One of the strings to score against.
 26  * @param {String} str2 One of the strings to score against.
 27  * @returns {Number}    Returns the computed Levenshtein distance between the two argument strings.
 28  * @extends             String
 29  */
 30 String.levenshteinDistance = function(str1, str2) {
 31     if(str1.length === 0) return str2.length;
 32     if(str2.length === 0) return str1.length;
 33 
 34     var matrix = [];
 35 
 36     // increment along the first column of each row
 37     var i;
 38     for(i = 0; i <= str2.length; i++){
 39         matrix[i] = [i];
 40     }
 41 
 42     // increment each column in the first row
 43     var j;
 44     for(j = 0; j <= str1.length; j++){
 45         matrix[0][j] = j;
 46     }
 47 
 48     // Fill in the rest of the matrix
 49     for(i = 1; i <= str2.length; i++){
 50         for(j = 1; j <= str1.length; j++){
 51             if(str2.charAt(i-1) == str1.charAt(j-1)){
 52                 matrix[i][j] = matrix[i-1][j-1];
 53             } else {
 54                 matrix[i][j] = Math.min(matrix[i-1][j-1] + 1, // substitution
 55                                                                 Math.min(matrix[i][j-1] + 1, // insertion
 56                                                                                  matrix[i-1][j] + 1)); // deletion
 57             }
 58         }
 59     }
 60 
 61     return matrix[str2.length][str1.length];
 62 };
