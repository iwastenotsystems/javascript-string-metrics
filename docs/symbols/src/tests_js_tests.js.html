1 test('String.levenshteinDistance()', function() {

 10     ok(String.levenshteinDistance('aa', 'A') === 2, "'aa', 'A' <>= 1");

 12     ok(String.levenshteinDistance('Saturday', 'Sunday') === 3, "'Saturday', 'Sunday' <>= 3");
